Media Feature - width, min-width, max-width

width

@media (width: 800px) { body::before { display: block; content: 'The viewport width is exactly 800px'; } }

min-width

@media (min-width: 800px) { body::before { display: block; content: 'The viewport width is larger than 800px'; } }

max-width

@media (max-width: 800px) { body::before { display: block; content: 'The viewport width is larger than 800px'; } }
